export type V3DensityMode = 'linear' | 'compact' | 'batch' | 'archive'
export function densityForCount(count: number): V3DensityMode {

  if (count <= 19) return 'compact'
  if (count <= 99) return 'batch'
  return 'archive'
}

#### `src/lib/upload/v3-state.ts` (new)

Exports:
- `V3Action` (re-export from v3-types for ergonomics)
- `v3InitialState(batchId: string): V3State`
- `v3Reducer(state: V3State, action: V3Action): V3State` — pure function, exhaustive switch over V3Action

Reducer requirements:
- Exhaustive switch with TypeScript `never` check at the default case
- Pure (no side effects, no randomness, no current-time except where explicitly part of action payload)
- For every action in `C2-PLAN.md` §5.2 (KEEP) and §5.3 (NEW), produce the documented state transition
- Reject malformed actions with explicit guards (e.g., `BULK_ACCEPT_PROPOSALS_FOR_GROUP` with `fields` array containing `'price'` MUST be rejected — throw `Error('bulk_accept_price_forbidden')` per spec §9.2)

Key transitions (non-exhaustive — full list derived from §5):
- `OPEN_SIDE_PANEL`: sets `ui.sidePanelOpenAssetId`; preserves selection
- `CLOSE_SIDE_PANEL`: clears `ui.sidePanelOpenAssetId`; preserves selection
- `NAVIGATE_SIDE_PANEL`: moves `ui.sidePanelOpenAssetId` to next/prev within filtered visible assets per IP-8
- `BEGIN_COMMIT`: `commit.phase: 'idle' → 'summary'`; allows further field edits per IP-9 (no UI lock)
- `CANCEL_COMMIT`: `commit.phase: 'summary' → 'idle'`
- `CONFIRM_COMMIT`: `commit.phase: 'summary' → 'committing'`; initializes `perAssetProgress = {}`
- `COMMIT_PROGRESS_UPDATE`: updates `commit.perAssetProgress[assetId]`
- `COMMIT_SUCCEEDED`: `commit.phase: 'committing' → 'success'`
- `COMMIT_PARTIALLY_FAILED`: `commit.phase: 'committing' → 'partial-failure'`; sets `commit.failed`
- `RETRY_FAILED_COMMITS`: `commit.phase: 'partial-failure' → 'committing'`; clears `commit.failed`; reinitializes per-asset progress for failed-only
- `RESET_FLOW`: returns to `v3InitialState(state.batch.id)`
- `ACCEPT_PROPOSAL`: NO STATE CHANGE per IP-5 (selector-derived); kept as a no-op action for telemetry hooks (or omitted — see §6 IP)
- `REGENERATE_PROPOSAL`: dispatches a re-fetch (mock simulation engine integration in C2.5); for C2.1, a no-op
- `RECEIVE_AI_CLUSTER_PROPOSAL`: appends to `aiClusterProposals` with `status: 'pending'`
- `ACCEPT_AI_CLUSTER_PROPOSAL`: marks proposal `status: 'accepted'`; creates Story group from proposed asset IDs
- `DISMISS_AI_CLUSTER_PROPOSAL`: marks proposal `status: 'dismissed'`
